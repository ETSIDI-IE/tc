\documentclass{standalone}
\usepackage{mathpazo}

\usepackage{circuitikz}
\usetikzlibrary{calc}

\begin{document}

\begin{circuitikz}[european resistors, american currents, american voltages]
  \coordinate (Ag) at (0,4);
  \coordinate (Bg) at (0,0);
  \node[label=A] (A) at ($(Ag) + (1,0)$) {};
  \node[label=below:B] (B) at ($(Bg) + (1,0)$) {};
  \draw
  (Bg) to [isource, l= $I_g$] (Ag)
  (Ag) to [short, i = $I$] (A)
  (Bg) to [short] (B)
  (A) to [open, *-*, v^= $U_{AB}$] (B);
\end{circuitikz}

\end{document}
